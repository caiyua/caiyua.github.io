    Element-Ui使用

    11/22/2021

    # 安装

    推荐使用 npm 的方式安装，它能更好地和 webpack (opens new window) 打包工具配合使用。

    npm i element-ui -S
    1

    # 完整引入

    在main.js文件中写入

    import ElementUI from 'element-ui'
    1
    import 'element-ui/lib/theme-chalk/index.css'
    1
    Vue.use(ElementUI)
    1

    # 按需引入

    命令行中输入

    vue add element
    1

    选择按需引入，Import on demand，然后选择zh-CN

    这时候多了一个plugins的文件夹，里面有一个element.js的文件，并且main.js中多了一个import的内容，babel.config.js也注入了相关的内容。

    然后我们就可以直接在页面直接使用了，使用element的哪一个组件的话，都需要在element.js文件进行引入、使用的。

    # Layout 布局

    通过基础的 24 分栏，迅速简便地创建布局。

          <el-row :gutter="20">
      
        <el-col :span="8">
        	<el-button type="primary">内容1</el-button>
        </el-col>
        
        <el-col :offset="14" :span="2">
          <el-button type="primary">内容2</el-button>
        </el-col>
        
      </el-row>

:span="10"   // 给每一列设置的份数，一行共有 24份，

:offset="10" // 偏移量，从左往右偏移多少份

    # 表单

    	<el-form
        :model="loginForm"
        :rules="rules"
        ref="loginRef"
      >
        <!-- 用户名 -->
        <el-form-item prop="username">
          <el-input
            placeholder="用户名"
            v-model="loginForm.username"
          ></el-input>
        </el-form-item>
        <!-- 密码 -->
        <el-form-item prop="password">
          <el-input
            placeholder="密码"
            type="password"
            v-model="loginForm.password"
            show-password
          ></el-input>
        </el-form-item>
        <!-- 按钮区 -->
        <el-form-item>
          <el-button
            type="primary"
            class="loginBtn"
            @click="login"
          >安全登录</el-button>
        </el-form-item>
      </el-form>
